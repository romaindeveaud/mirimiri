    def initialize(corpus,mem="1g",count="1000",offset="1",run_id="default",print_query=false,print_docs=false)
      @index_path  = corpus
      @memory      = mem
      @count       = count
      @offset      = offset
      @run_id      = run_id
      @print_query = print_query ? "true" : "false"
      @print_docs  = print_docs  ? "true" : "false"
    end
